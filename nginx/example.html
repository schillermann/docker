Proxy docker container works!

This is the default web page for nginx server.

Create docker-compose.yml file in your project folder with the following content, to generate containers.


version: '2.0'
services:
  proxy:
    image: nginx:latest
    ports:
      - 80:80
    volumes:
      - ./public:/app/public
    networks:
      - bridge
  app:
    image: php7:latest
    volumes:
      - ./:/app
    networks:
      - bridge
networks:
  bridge:
    driver: bridge
